# the difference in ASCII values between 122 (z) and 65 (A)
# this is used as the loop index
seti $0 57

# ascii value of A, when added to $0 yields the current char we are printing
seti $1 65

# loop until $0 is 1
sloop $0

# copy the current loop index into $2, add 65
mov $0 $2
addi $2 65
# and print
printc $2

# subtract 1 from $0
subi $0 1
eloop $0


# print a newline
seti $0 10
printc $0
